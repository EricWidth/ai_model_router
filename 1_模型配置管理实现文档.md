# 模型配置管理实现文档

## 一、功能概述

模型配置管理是 AI Model Router 的核心功能之一，负责管理所有模型配置的加载、验证、存储和查询。

## 二、核心职责

1. 配置文件加载与解析（支持 YAML/JSON 格式）
2. 配置项验证（使用 AJV JSON Schema）
3. 环境变量替换（支持 `${VAR}` 语法）
4. 配置热更新（可选）
5. 配置持久化存储（SQLite/JSON）

## 五、实现细节

### 5.1 配置文件加载

```typescript
// src/config/loader.ts
import yaml from 'js-yaml'
import * as fs from 'fs/promises'

export class ConfigLoader {
  constructor(private filePath: string) {}

  async load(): Promise<any> {
    const content = await fs.readFile(this.filePath, 'utf-8')
    const extension = path.extname(this.filePath)

    switch (extension) {
      case '.yaml':
      case '.yml':
        return yaml.load(content)
      case '.json':
        return JSON.parse(content)
      default:
        throw new Error(`Unsupported config format: ${extension}`)
    }
  }
}
```

### 5.2 环境变量处理

```typescript
// src/config/env.ts
export class EnvProcessor {
  process(obj: any): any {
    if (typeof obj === 'string') {
      return this.replaceEnvVars(obj)
    }

    if (Array.isArray(obj)) {
      return obj.map(item => this.process(item))
    }

    if (typeof obj === 'object' && obj !== null) {
      const result: any = {}
      for (const [key, value] of Object.entries(obj)) {
        result[key] = this.process(value)
      }
      return result
    }

    return obj
  }

  private replaceEnvVars(value: string): string {
    return value.replace(/\$\{([^}]+)\}/g, (_, varName) => {
      const envValue = process.env[varName]
      if (envValue === undefined) {
        throw new Error(`Environment variable ${varName} not found`)
      }
      return envValue
    })
  }
}
```

### 5.4 配置管理器

```typescript
// src/config/index.ts
import Ajv from 'ajv'
import { ConfigLoader } from './loader'
import { EnvProcessor } from './env'
import { ConfigSchema } from './schema'
import type { Config } from '../types'

export class ConfigManager {
  private loader: ConfigLoader
  private envProcessor: EnvProcessor
  private ajv: Ajv
  private config: Config | null = null

  constructor(filePath: string) {
    this.loader = new ConfigLoader(filePath)
    this.envProcessor = new EnvProcessor()
    this.ajv = new Ajv({ allErrors: true })
  }

  async load(): Promise<Config> {
    const rawConfig = await this.loader.load()
    const processedConfig = this.envProcessor.process(rawConfig)

    const validate = this.ajv.compile(ConfigSchema)
    const valid = validate(processedConfig)

    if (!valid) {
      throw new Error(`Config validation failed: ${JSON.stringify(validate.errors, null, 2)}`)
    }

    this.config = processedConfig as Config
    return this.config
  }

  get<K extends keyof Config>(key: K): Config[K] {
    if (!this.config) {
      throw new Error('Config not loaded')
    }
    return this.config[key]
  }

  getConfig(): Config {
    if (!this.config) {
      throw new Error('Config not loaded')
    }
    return this.config
  }

  async reload(): Promise<void> {
    await this.load()
  }
}
```

## 六、配置持久化

### 6.1 SQLite 存储

```typescript
// src/config/storage.ts
import Database from 'better-sqlite3'

export class ConfigStorage {
  private db: Database.Database

  constructor(dbPath: string) {
    this.db = new Database(dbPath)
    this.initTables()
  }

  private initTables(): void {
    this.db.exec(`
      CREATE TABLE IF NOT EXISTS models (
        id TEXT PRIMARY KEY,
        name TEXT NOT NULL,
        provider TEXT NOT NULL,
        api_key TEXT NOT NULL,
        base_url TEXT,
        quota INTEGER,
        priority INTEGER NOT NULL,
        enabled INTEGER DEFAULT 1,
        created_at INTEGER,
        updated_at INTEGER
      )
    `)
  }

  async saveModel(model: ModelConfig): Promise<void> {
    const stmt = this.db.prepare(`
      INSERT OR REPLACE INTO models
      (id, name, provider, api_key, base_url, quota, priority, enabled, created_at, updated_at)
      VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
    `)
    stmt.run(
      model.name,
      model.name,
      model.provider,
      model.apiKey,
      model.baseUrl || null,
      model.quota || null,
      model.priority,
      model.enabled ? 1 : 0,
      Date.now(),
      Date.now()
    )
  }

  async getModels(): Promise<ModelConfig[]> {
    const stmt = this.db.prepare('SELECT * FROM models')
    const rows = stmt.all()
    return rows.map(row => ({
      name: row.name,
      provider: row.provider,
      apiKey: row.api_key,
      baseUrl: row.base_url,
      quota: row.quota,
      priority: row.priority,
      enabled: row.enabled === 1
    }))
  }

  async deleteModel(name: string): Promise<void> {
    const stmt = this.db.prepare('DELETE FROM models WHERE name = ?')
    stmt.run(name)
  }
}
```
