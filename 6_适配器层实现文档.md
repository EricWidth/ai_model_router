# 适配器层实现文档

## 一、功能概述

适配器层是 AI Model Router 的核心抽象层，负责将不同云服务商的 API 统一为标准接口，支持 OpenAI、阿里云等多家服务商。

## 二、核心职责

1. 定义统一的模型接口
2. 实现各服务商的适配器
3. 处理不同服务商的特殊逻辑
4. 统一错误处理和重试逻辑
5. 管理请求限流和配额

## 五、实现细节

### 5.1 基础适配器

```typescript
// src/adapters/base.ts
import axios, { AxiosInstance, AxiosRequestConfig } from 'axios'
import { ChatCompletionRequest, ChatCompletionResponse, AudioSpeechRequest, ImageGenerationResponse, ModelType } from '../types'

export abstract class BaseAdapter implements ModelAdapter {
  protected client: AxiosInstance
  protected healthCheckEnabled: boolean = true

  constructor(
    public readonly name: string,
    public readonly provider: string,
    protected apiKey: string,
    protected baseUrl: string,
    protected timeout: number = 30000
  ) {
    this.client = axios.create({
      baseURL: baseUrl,
      timeout,
      headers: {
        'Authorization': `Bearer ${apiKey}`,
        'Content-Type': 'application/json'
      }
    })
  }

  abstract chat(request: ChatCompletionRequest): Promise<ChatCompletionResponse>
  abstract speech(request: AudioSpeechRequest): Promise<Buffer>
  abstract image(request: ImageGenerationRequest): Promise<ImageGenerationResponse>
  abstract getType(): ModelType

  async healthCheck(): Promise<void> {
    if (!this.healthCheckEnabled) {
      return
    }

    try {
      await this.chat({
        messages: [{ role: 'user', content: 'ping' }],
        max_tokens: 1
      })
    } catch (error) {
      throw new Error(`Health check failed for ${this.name}: ${error}`)
    }
  }

  protected async request(config: AxiosRequestConfig): Promise<any> {
    try {
      const response = await this.client.request(config)
      return response.data
    } catch (error) {
      if (axios.isAxiosError(error)) {
        if (error.response?.status === 401) {
          throw new Error('Invalid API key')
        } else if (error.response?.status === 429) {
          throw new Error('Rate limit exceeded')
        } else if (error.code === 'ECONNABORTED') {
          throw new Error('Request timeout')
        } else {
          throw new Error(error.response?.data?.error?.message || error.message)
        }
      }
      throw error
    }
  }

  protected normalizeError(error: any): Error {
    if (error.response) {
      const data = error.response.data
      const message = data?.error?.message || data?.message || error.message

      if (error.response.status === 401) {
        return new Error(`Authentication failed: ${message}`)
      } else if (error.response.status === 429) {
        return new Error(`Rate limit exceeded: ${message}`)
      } else if (error.response.status === 500) {
        return new Error(`Server error: ${message}`)
      }

      return new Error(message)
    }

    return error
  }
}
```

### 5.2 OpenAI 适配器

```typescript
// src/adapters/openai.ts
import { BaseAdapter } from './base'
import { ChatCompletionRequest, ChatCompletionResponse, AudioSpeechRequest, ImageGenerationRequest, ImageGenerationResponse, ModelType } from '../types'

export class OpenAIAdapter extends BaseAdapter {
  getType(): ModelType {
    if (this.name.startsWith('tts-')) {
      return 'voice'
    } else if (this.name.startsWith('dall-e') || this.name.startsWith('gpt-image')) {
      return 'image'
    }
    return 'text'
  }

  async chat(request: ChatCompletionRequest): Promise<ChatCompletionResponse> {
    const response = await this.request({
      method: 'POST',
      url: '/chat/completions',
      data: {
        model: this.name,
        ...request
      }
    })

    return response
  }

  async speech(request: AudioSpeechRequest): Promise<Buffer> {
    const response = await this.request({
      method: 'POST',
      url: '/audio/speech',
      responseType: 'arraybuffer',
      data: {
        model: this.name,
        ...request
      }
    })

    return Buffer.from(response)
  }

  async image(request: ImageGenerationRequest): Promise<ImageGenerationResponse> {
    const response = await this.request({
      method: 'POST',
      url: '/images/generations',
      data: {
        model: this.name,
        ...request
      }
    })

    return response
  }
}
```

### 5.5 适配器工厂

```typescript
// src/adapters/factory.ts
import { OpenAIAdapter } from './openai'
import { AliyunAdapter } from './aliyun'
import { CustomAdapter } from './custom'
import { ModelAdapter } from './base'
import { ModelConfig } from '../types'

export function createAdapter(config: ModelConfig): ModelAdapter {
  const { name, provider, apiKey, baseUrl, timeout, type } = config

  switch (provider) {
    case 'openai':
      return new OpenAIAdapter(name, apiKey, baseUrl || 'https://api.openai.com/v1', timeout)

    case 'aliyun':
      return new AliyunAdapter(
        name,
        apiKey,
        baseUrl || 'https://dashscope.aliyuncs.com/compatible-mode/v1',
        timeout
      )

    case 'anthropic':
      return new AnthropicAdapter(name, apiKey, baseUrl || 'https://api.anthropic.com/v1', timeout)

    case 'custom':
      return new CustomAdapter(name, apiKey, baseUrl, timeout, type)

    default:
      throw new Error(`Unsupported provider: ${provider}`)
  }
}

export function createAdapters(configs: ModelConfig[]): Map<string, ModelAdapter> {
  const adapters = new Map<string, ModelAdapter>()

  for (const config of configs) {
    try {
      const adapter = createAdapter(config)
      adapters.set(config.name, adapter)
    } catch (error) {
      console.error(`Failed to create adapter for ${config.name}:`, error)
    }
  }

  return adapters
}
```

### 5.6 Anthropic 适配器

```typescript
// src/adapters/anthropic.ts
import { BaseAdapter } from './base'
import { ChatCompletionRequest, ChatCompletionResponse, AudioSpeechRequest, ImageGenerationRequest, ImageGenerationResponse, ModelType } from '../types'

export class AnthropicAdapter extends BaseAdapter {
  constructor(
    name: string,
    apiKey: string,
    baseUrl: string = 'https://api.anthropic.com/v1',
    timeout: number = 30000
  ) {
    super(name, 'anthropic', apiKey, baseUrl, timeout)
    this.client.defaults.headers = {
      'x-api-key': apiKey,
      'anthropic-version': '2023-06-01',
      'Content-Type': 'application/json'
    }
  }

  getType(): ModelType {
    return 'text'
  }

  private convertMessages(request: ChatCompletionRequest): any {
    const messages = request.messages.map(msg => {
      if (msg.role === 'system') {
        return { role: msg.role, content: msg.content }
      }
      return { role: msg.role, content: msg.content }
    })

    return {
      model: this.name,
      messages,
      temperature: request.temperature,
      top_p: request.top_p,
      max_tokens: request.max_tokens || 1000,
      stream: request.stream || false
    }
  }

  private convertResponse(response: any): ChatCompletionResponse {
    return {
      id: response.id,
      object: 'chat.completion',
      created: Date.now(),
      model: this.name,
      choices: response.content.map((content: any, index: number) => ({
        index,
        message: {
          role: 'assistant',
          content: content.text
        },
        finish_reason: response.stop_reason
      })),
      usage: {
        prompt_tokens: response.usage?.input_tokens || 0,
        completion_tokens: response.usage?.output_tokens || 0,
        total_tokens: (response.usage?.input_tokens || 0) + (response.usage?.output_tokens || 0)
      }
    }
  }

  async chat(request: ChatCompletionRequest): Promise<ChatCompletionResponse> {
    const body = this.convertMessages(request)
    const response = await this.request({
      method: 'POST',
      url: '/messages',
      data: body
    })

    return this.convertResponse(response)
  }

  async speech(_request: AudioSpeechRequest): Promise<Buffer> {
    throw new Error('Anthropic does not support speech synthesis')
  }

  async image(_request: ImageGenerationRequest): Promise<ImageGenerationResponse> {
    throw new Error('Anthropic does not support image generation')
  }
}
```

## 六、使用示例

### 6.1 创建适配器

```typescript
import { createAdapter } from './adapters/factory'

const openaiAdapter = createAdapter({
  name: 'gpt-4',
  provider: 'openai',
  apiKey: '<YOUR_OPENAI_API_KEY>',
  baseUrl: 'https://api.openai.com/v1',
  timeout: 30000
})

const response = await openaiAdapter.chat({
  messages: [{ role: 'user', content: 'Hello' }]
})
```

### 6.2 批量创建适配器

```typescript
import { createAdapters } from './adapters/factory'

const configs = [
  {
    name: 'gpt-4',
    provider: 'openai',
    apiKey: '<YOUR_OPENAI_API_KEY>',
    baseUrl: 'https://api.openai.com/v1',
    priority: 1
  },
  {
    name: 'qwen-max',
    provider: 'aliyun',
    apiKey: '<YOUR_ALIYUN_API_KEY>',
    priority: 2
  }
]

const adapters = createAdapters(configs)

for (const [name, adapter] of adapters.entries()) {
  console.log(`Created adapter for ${adapter.provider}/${name}`)
}
```

### 6.3 在路由中使用

```typescript
import { createAdapters } from './adapters/factory'

const configs = [...config.models.text, ...config.models.voice, ...config.models.image]
const adapters = createAdapters(configs)

router.post('/v1/chat/completions', async (req, res) => {
  const result = await switchStrategy.execute('text', async (modelName) => {
    const adapter = adapters.get(modelName)
    if (!adapter) {
      throw new Error(`Adapter not found for model: ${modelName}`)
    }
    return await adapter.chat(req.body)
  })

  res.json(result)
})
```
